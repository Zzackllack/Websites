Solid Projekt

Solid gibt Nutzern die volle Kontrolle über ihre Daten. Speichere Daten in „Pods“ und teile sie nur nach Bedarf.

Was ist Solid?

  • Dezentrale Plattform zur Datenspeicherung.
  • Daten werden in persönlichen „Pods“ gespeichert.
  • Nutzer kontrollieren, wer Zugang hat.

Vorteile von Solid

  • Volle Datenkontrolle.
  • Verbesserter Datenschutz.
  • Hohe Datensicherheit.

Anwendungsbeispiele

  • Soziale Netzwerke.
  • Gesundheitsanwendungen.
  • Persönliche Datenverwaltung.

Nachteile

  • Komplexe Einrichtung für Laien.
  • Begrenzte Unterstützung durch Apps und Dienste.
  • Erfordert technische Kenntnisse für volle Nutzung.
  • Abhängigkeit von Hosting-Anbietern für Pods.

Partner

  • Inrupt: Förderung und Verbreitung.
  • MIT: Forschung und Entwicklung.
  • BBC: Datengetriebene Experimente.
  • Redecentralize.org: Dezentralisierung von Daten.

Zukunft von Solid

  • Transformiert das Internet.
  • Erhöhte Akzeptanz erwartet.
  • Teil des zukünftigen Internets.

// Realistisches Beispiel: Zugriff auf einen Solid-Pod mit JavaScript
import { getSolidDataset, getThing, getStringNoLocale } from "@inrupt/solid-client";
import { Session } from "@inrupt/solid-client-authn-browser";

const podUrl = "https://example.pod.provider/user1/profile/card";
const session = new Session();

async function fetchData() {
    try {
        await session.login({
            oidcIssuer: "https://provider.example",
            clientName: "Meine Solid App",
            redirectUrl: window.location.href,
        });

        const dataset = await getSolidDataset(podUrl, { fetch: session.fetch });
        const profile = getThing(dataset, `${podUrl}#me`);
        const fullName = getStringNoLocale(profile, "http://xmlns.com/foaf/0.1/name");

        console.log(`Name des Nutzers: ${fullName}`);
    } catch (error) {
        console.error("Fehler beim Abrufen der Daten:", error);
    }
}

fetchData();

// Realistisches Beispiel: Zugriff auf einen Solid-Pod mit Go
package main

import (
    "fmt"
    "io/ioutil"
    "net/http"
)

const podURL = "https://example.pod.provider/user1/profile/card"

func fetchData() {
    req, _ := http.NewRequest("GET", podURL, nil)
    req.Header.Add("Authorization", "Bearer your-access-token")

    client := &http.Client{}
    resp, err := client.Do(req)

    if err != nil {
        fmt.Println("Fehler beim Abrufen der Daten:", err)
        return
    }

    defer resp.Body.Close()
    body, _ := ioutil.ReadAll(resp.Body)

    if resp.StatusCode == 200 {
        fmt.Println("Daten erfolgreich abgerufen:", string(body))
    } else {
        fmt.Println("Fehler beim Abrufen der Daten:", resp.StatusCode)
    }
}

func main() {
    fetchData()
}
                
            
                
# Realistisches Beispiel: Zugriff auf einen Solid-Pod mit Ruby
require 'net/http'
require 'uri'

pod_url = "https://example.pod.provider/user1/profile/card"
auth_token = "Bearer your-access-token"

uri = URI.parse(pod_url)
request = Net::HTTP::Get.new(uri)
request["Authorization"] = auth_token

response = Net::HTTP.start(uri.host, uri.port, use_ssl: true) do |http|
    http.request(request)
end

if response.code.to_i == 200
    puts "Daten erfolgreich abgerufen: #{response.body}"
else
    puts "Fehler beim Abrufen der Daten: #{response.code}"
end
